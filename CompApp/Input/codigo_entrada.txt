public class Codigo01 {
    public static void main ( String [ ] args ){
        double n, a, b;
        n = 10;
        a = 0;
        b = 1;
        printFibonacci(n, a, b);
    }
    public static double printFibonacci(double n, double a, double  b){
        if (n > 0) {
            System.out.println(a);
            n = n-1;
            a = a + b;
            printFibonacci(n, b, a);
        }
        return 0.0;
    }
}
